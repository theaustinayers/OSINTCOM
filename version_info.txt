# UTF-8
#
# For more details about fixed file info 'ffi' see:
# http://msdn.microsoft.com/en-us/library/ms646997.aspx!
VSVersionInfo(
  ffi=VSFixedFileInfo(
    # Contains as much info as Windows will allow. None of the fields can be empty.
    mask=0x3f,
    mask_ex=0x3f,
    # Contains a bitmask that specifies the valid bits 'flags'r
    reserved=0x0,
    serial=0x0,
    # Contains a bitmask that specifies the Boolean attributes of the file.
    # The STRFILETYPE comments show all the bits that can be used.
    strFileType="VFT_APP",
    # Specifies the function of the file.
    # STRFILETYPE_COMMENTS shows all the possible values
    strFileSubtype="VFT2_UNKNOWN",
    # Contains any attributes for this file.  For this example it contains NT file attributes
    mask &= ~0x4) # UNIC returned in index 1, is not needed.
  kids=None,
  timestamp=None,
  version=None
)
